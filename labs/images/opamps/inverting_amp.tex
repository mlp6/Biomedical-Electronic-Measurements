\begin{circuitikz}
    % Draw input resistor Ri
    \draw (-1,2) node[above] {$V_{in}$} to[R, l=$R_i$, o-] (2,2);
    % Draw op-amp
    \draw (2,2) node[op amp, anchor=-] (opamp) {};
    % Add power supplies
    \draw (opamp.up) to[short] (opamp.up -| 3.1,3.25) node[vcc] {+12 V};
    \draw (opamp.down) to[C, l=$0.1 uF$] ($(opamp.down)+(0,-2)$) -- ($(opamp.down)+(0,-1.5)$) node[vee] {-12 V};
    
    % to[short] (opamp.down -| 3.1,0) node[vee] {-12 V};
    
    % Draw feedback resistor Rf
    \draw (opamp.out) -- (4.4,5) to[R, l=$R_f$] (2,5) to[short] (opamp.-);

    \draw (opamp.+) to[short] (2,0) node[ground] {};
    % Draw output
    \draw (opamp.out) to[short, -o] (opamp.out -| 5,0) node[above] {$V_{out}$};
