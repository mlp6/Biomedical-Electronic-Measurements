\documentclass{standalone}
\usepackage[siunitx,american,RPvoltages]{circuitikz}
\begin{document}
\begin{circuitikz}
    % Draw input resistor Ri
    \draw (-1,2) node[above] {$V_{in}$} to[R, l=$R_i$, o-] (2,2);
    % Draw op-amp
    \draw (2,2) node[op amp, anchor=-] (opamp) {};
    % Add power supplies
    \draw (opamp.up) -- ($(opamp.up)+(0,0.3)$) node[vcc] {+12 V};
    \draw (opamp.down) -- ($(opamp.down)+(0,-0.3)$) node[vee] {-12 V};
    
    % Draw feedback resistor Rf
    \draw (opamp.out) to[short] ($(opamp.out)+(1.1,0)$) -- (5.5,4.5) to[R, l=$R_f$] (2,4.5) to[short] (opamp.-);
    % Draw ground to non-inverting input
    \draw (opamp.+) to[short] (2,0) node[ground] {};
    % Draw output
    \draw (opamp.out) to[short, -o] ($(opamp.out)+(2,0)$) node[above] {$V_{out}$};
    
    % Add bypass capacitors
    \draw (9,3) node[vcc] {+12 V} 
        to[C, l_=$0.1\,\mu F$, -] (9,2)
        to[C, l_=$0.1\,\mu F$, -] (9,1) 
        to[short] (9,1) node[vee] {-12 V};
    \draw (9,2) -- (10,2) -- (10,1.5) node[ground] {};
\end{circuitikz}
\end{document}
